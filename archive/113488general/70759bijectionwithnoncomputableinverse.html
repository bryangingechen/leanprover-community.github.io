def enat := {f : nat → bool // ∀ n, f n = tt → f (n+1) = tt}

def f : nat → enat
| 0 := ⟨λ _, ff, λ n h, h⟩
| (n+1) := ⟨λ i, n ≤ i, λ i h,
  (to_bool_iff _).2 $ nat.le_succ_of_le $ (to_bool_iff _).1 h⟩

the inverse to f is noncomputable
